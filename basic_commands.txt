#Basic commands for Git

git init			//Initialize Local Git Repository
git add <file>			//Add File(s) To Index
git add .			//Add All Files To Index
git status 			//Check Status Of Working Tree
git commit			//Commit Changes In Index
git commit -m "message"		//Commit Changes withou open a commit tex to leave a message

git push			//Push To Remote Repository
git pull			//Pull Laest From Remote Repository
git clone			//Clone Repository Into A New Diretory
    				//Nomally used to copy a remote directoryinto your current folder


git status			//Check the status
git rm --cached <file>		//romove the staged file to unstage


git branch <branchname>		//creat a new branch
git checkout <branchname>	//switched to branch <branchname>
git chckout master		//switched to branch master,wich means main branch

git merge <branchname>
